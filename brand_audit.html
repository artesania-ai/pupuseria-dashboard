        // Data

        const actionPlan = [

        const keywords = [

            {

            },

            {

                keyword: "Pupusas delivery San Francisco",

                competition: "Medium",

            },

            {

                keyword: "Traditional Central American restaurant",

                competition: "Medium",

                opportunity: "Blog content about Salvadoran food culture and traditions"

            }

        ];


        // State

        let completedTasks = new Set();

        let currentDay = 1;


        // Load saved data

        function loadData() {

            const saved = localStorage.getItem('pupuseria-progress');

            if (saved) {

                try {

                    const data = JSON.parse(saved);

                    completedTasks = new Set(data.completedTasks || []);

                    currentDay = data.currentDay || 1;

                    document.getElementById('day-slider').value = currentDay;

                    updateCurrentDay(currentDay);

                } catch (e) {

                    console.log('Error loading saved data:', e);

                }

            }

            updateProgress();

        }


        // Save data

        function saveData() {

            const data = {

                completedTasks: [...completedTasks],

                currentDay: currentDay

            };

            localStorage.setItem('pupuseria-progress', JSON.stringify(data));

        }


        // Update progress

        function updateProgress() {

            const percentage = Math.round((completedTasks.size / actionPlan.length) * 100);

            document.getElementById('progress-percentage').textContent = percentage + '%';

            document.getElementById('progress-fill').style.width = percentage + '%';

            document.getElementById('progress-text').textContent = 

                `${completedTasks.size} of ${actionPlan.length} tasks completed`;

        }


        // Toggle task completion

        function toggleTask(day) {

            if (completedTasks.has(day)) {

                completedTasks.delete(day);

            } else {

                completedTasks.add(day);

            }

            updateProgress();

            renderTasks();

            saveData();

        }


        // Update current day

        function updateCurrentDay(day) {

            currentDay = parseInt(day);

            document.getElementById('current-day-display').textContent = `Day ${currentDay} of 30`;

            renderTasks();

            saveData();

        }


        // Render tasks

        function renderTasks() {

            const container = document.getElementById('tasks-container');

            container.innerHTML = '';


            actionPlan.forEach(task => {

                const isCompleted = completedTasks.has(task.day);

                const isCurrent = task.day <= currentDay;

                

                const taskElement = document.createElement('div');

                taskElement.className = `task-item ${isCompleted ? 'completed' : ''} ${isCurrent && !isCompleted ? 'current' : ''}`;

                taskElement.onclick = () => toggleTask(task.day);

                

                taskElement.innerHTML = `

                    <div class="checkbox ${isCompleted ? 'checked' : ''}" onclick="event.stopPropagation(); toggleTask(${task.day})">

                        ${isCompleted ? '<i data-lucide="check" style="width: 12px; height: 12px;"></i>' : ''}

                    </div>

                    <div style="flex: 1;">

                        <div style="display: flex; align-items: center; gap: 8px; margin-bottom: 4px;">

                            <span style="font-weight: 600; font-size: 14px;">Day ${task.day}</span>

                            <span class="priority-badge priority-${task.priority}">${task.priority}</span>

                        </div>

                        <p style="font-size: 14px; color: ${isCompleted ? '#6b7280' : '#1f2937'};">

                            ${task.action}

                        </p>

                    </div>

                `;

                

                container.appendChild(taskElement);

            });


            // Re-initialize Lucide icons

            lucide.createIcons();

        }


        // Render keywords

        function renderKeywords() {

            const container = document.getElementById('keywords-container');

            container.innerHTML = '';


            keywords.forEach((item, index) => {

                const keywordElement = document.createElement('div');

                keywordElement.className = 'keyword-card';

                

                keywordElement.innerHTML = `

                    <div style="display: flex; justify-content: space-between; align-items: flex-start; margin-bottom: 12px;">

                        <div>

                            <h4 style="font-weight: 600; margin-bottom: 8px;">"${item.keyword}"</h4>

                            <span class="competition-badge competition-${item.competition.toLowerCase()}">${item.competition} Competition</span>

                        </div>

                        <i data-lucide="chevron-right" style="width: 16px; height: 16px; color: #9ca3af;"></i>

                    </div>

                    <p style="font-size: 14px; color: #6b7280;">

                        <span style="font-weight: 600;">Opportunity:</span> ${item.opportunity}

                    </p>

                `;

                

                container.appendChild(keywordElement);

            });


            // Re-initialize Lucide icons

            lucide.createIcons();

        }


        // Show tab

        function showTab(tabName) {

            // Hide all tabs

            document.querySelectorAll('.tab-content').forEach(tab => {

                tab.classList.add('hidden');

            });

            

            // Remove active class from all tab buttons

            document.querySelectorAll('.tab').forEach(tab => {

                tab.classList.remove('active');

            });

            

            // Show selected tab

            document.getElementById(tabName + '-tab').classList.remove('hidden');

            

            // Add active class to clicked tab button

            event.target.classList.add('active');

        }


        // Initialize the app

        function init() {

            loadData();

            renderTasks();

            renderKeywords();

            

            // Initialize Lucide icons

            lucide.createIcons();

        }


        // Run when page loads

        window.addEventListener('DOMContentLoaded', init);
